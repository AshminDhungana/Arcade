# Feature 2.2.1: Agent Platform Abstraction Layer - Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `IPlatformService` interface, OS-specific implementations (Windows first), and the factory function that selects the correct implementation at runtime. This is the foundation for all agent platform-specific operations.

**Architecture:** Define a clean TypeScript interface (`IPlatformService`) that abstracts all OS-specific operations (kiosk overlay, screenshots, system commands, auto-start) behind a single contract. Each OS provides an implementation. A factory selects the correct implementation via `process.platform` at runtime. The interface methods are async where they perform I/O (screenshot, restart, etc.) and synchronous where they only mutate internal state (timer updates).

**Tech Stack:** Electron (`BrowserWindow`, `desktopCapturer`), TypeScript 6.x, `sharp` (image processing), `systeminformation` (hardware info), `child_process`, `vitest` (testing; TDD required).

## Global Constraints

- The agent project root is `agent/` -- all paths are relative to this unless stated otherwise.
- Module system: `type: "module"` (ESM), `moduleResolution: "NodeNext"`.
- Electron `kiosk: true`, `alwaysOnTop: true`, `closable: false`, `devTools: false` for kiosk overlay.
- Screenshot: JPEG at 80% quality, max 1280x720, returned as `Buffer`.
- Windows auto-start: registry `HKCU\Software\Microsoft\Windows\CurrentVersion\Run`.
- Linux/macOS auto-start: deferred to later phases.
- No OS lock/unlock methods -- the kiosk overlay is the access control mechanism.

```typescript
/**
 * Content displayed on the kiosk overlay when shown.
 */
export interface OverlayContent {
  /** Optional base64-encoded cafe logo or absolute file path. */
  cafeLogo?: string;

  /** Displayed as the header / cafe name. */
  cafeName: string;

  /** Active announcements (e.g. "Tournament at 6 PM"). */
  announcements: string[];

  /** Whether the "Call Staff" button is enabled. */
  callStaffEnabled: boolean;

  /** True if a session is currently active. */
  sessionActive: boolean;

  /** Remaining time as "HH:MM:SS" or "Unlimited" when absent. */
  remainingTime?: string;

  /** Whether to flash the low-time warning indicator. */
  lowTimeWarning?: boolean;
}

/**
 * Hardware / OS information returned by `getSystemInfo()`.
 */
export interface SystemInfo {
  /** CPU model name (e.g. "Intel(R) Core(TM) i7-9700K"). */
  cpuModel: string;

  /** Number of logical CPU cores. */
  cpuCores: number;

  /** Total system memory in gigabytes (rounded down). */
  totalMemoryGB: number;

  /** Total disk space in gigabytes (rounded down). */
  totalDiskGB: number;

  /** Operating system name (e.g. "Windows_NT"). */
  osName: string;

  /** OS version string (e.g. "10.0.22631"). */
  osVersion: string;

  /** Machine hostname. */
  hostname: string;
}

/**
 * Supported platform identifiers for the factory.
 */
export type PlatformName = 'win32' | 'darwin' | 'linux';

/**
 * Abstraction over all platform-specific operations.
 *
 * Implementations manage an internal `BrowserWindow` for the kiosk overlay
 * and provide OS-specific equivalents for screenshot capture, system
 * commands and auto-start registration.
 */
export interface IPlatformService {
  /**
   * Show (or create) the kiosk overlay window with the given content.
   *
   * Creates a new `BrowserWindow` with `kiosk: true`, `alwaysOnTop: true`,
   * `frame: false`, `closable: false`, `devTools: false`.
   *
   * Keyboard shortcuts (Alt+F4, F12, Ctrl+P, etc.) are intercepted and discarded
   * in the `before-input-event` handler.
   */
  showKioskOverlay(content: OverlayContent): void;

  /**
   * Hide and destroy the kiosk overlay window, if it exists.
   *
   * After hiding, the kiosk overlay is completely destroyed. A subsequent call
   * to `showKioskOverlay` will create a fresh window.
   */
  hideKioskOverlay(): void;

  /**
   * Update the visible timer display on the kiosk overlay.
   *
   * Must be called after `showKioskOverlay`. No-op if the overlay is not visible.
   */
  updateTimer(timeString: string): void;

  /**
   * Announce a message on the kiosk overlay for a given duration (milliseconds).
   *
   * No-op if the overlay is not visible.
   */
  sendAnnouncement(text: string, durationMs: number): void;

  /** Restart the PC immediately. */
  restartPC(): Promise<void>;

  /** Shut down the PC immediately. */
  shutdownPC(): Promise<void>;

  /**
   * Capture a screenshot of the primary display.
   *
   * Uses `desktopCapturer.getSources({ types: ['screen'] })` and resizes
   * the resulting image to 1280x720 max using `sharp` with 80% JPEG quality.
   */
  captureScreenshot(): Promise<Buffer>;

  /** Register the agent to start automatically on system boot. */
  enableAutoStart(): Promise<void>;

  /** Remove the auto-start registration. */
  disableAutoStart(): Promise<void>;

  /** Return hardware and OS details for the REGISTER payload. */
  getSystemInfo(): Promise<SystemInfo>;
}
```

```typescript
// agent/src/main/platform/index.ts
import type { IPlatformService } from './types.js';
export * from './types.js';

const PLATFORM_MAP: Record<string, string> = {
  win32: './windows.js',
  darwin: './macos.js',
  linux: './linux.js',
};

/**
 * Return the platform-specific IPlatformService implementation.
 *
 * Uses `process.platform` to determine the target module.
 * Only `win32` is implemented in this feature -- `darwin` and `linux`
 * are handled by Epic 7.1 and 7.2. This factory will call `require()`
 * dynamically so that only the relevant OS module is loaded at runtime.
 *
 * @throws {Error} if the current platform is not yet supported.
 */
export async function getPlatformService(): Promise<IPlatformService> {
  const platform = process.platform;

  if (!PLATFORM_MAP[platform]) {
    throw new Error(
      `Platform "${platform}" is not yet supported. Supported: ${Object.keys(PLATFORM_MAP).join(', ')}.`,
    );
  }

  if (platform === 'win32') {
    const { WindowsPlatformService } = await import('./windows.js');
    return new WindowsPlatformService();
  }

  // Fallback guard (unreachable in practice due to PLATFORM_MAP check above,
  // but keeps TypeScript exhaustive).
  throw new Error(`Platform "${platform}" is not yet supported.`);
}
```

```typescript
// agent/src/main/platform/windows.ts
import { BrowserWindow, desktopCapturer } from 'electron';
import { exec } from 'child_process';
import { promisify } from 'util';
import sharp from 'sharp';
import si from 'systeminformation';
import os from 'os';
import type { IPlatformService, OverlayContent, SystemInfo } from './types.js';

const execAsync = promisify(exec);

/**
 * Registry path used for Windows auto-start (                  \HKCU\Software\Microsoft\Windows\CurrentVersion\Run).
 */
const AUTO_START_KEY = 'HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Run';
const APP_NAME = 'ArcadeAgent';

/**
 * List of shortcuts that must be discarded while the kiosk overlay is active.
 */
const BLOCKED_SHORTCUTS = [
  'Alt+F4',
  'Alt+Shift+I',
  'Control+Shift+I',
  'Control+P',
  'F12',
];

/**
 * Windows implementation of `IPlatformService`.
 *
 * Manages a single `BrowserWindow` for the kiosk overlay and provides
 * OS-specific implementations for screenshots, restart, shutdown and
 * auto-start via the Windows registry.
 */
export class WindowsPlatformService implements IPlatformService {
  private kioskWindow: BrowserWindow | null = null;

  /** Number of times the window has shown (used for counter). */
  private showCount = 0;

  showKioskOverlay(content: OverlayContent): void {
    // Avoid recreating if already visible
    if (this.kioskWindow && !this.kioskWindow.isDestroyed()) {
      this.kioskWindow.show();
      this.kioskWindow.webContents.send('overlay:update', content);
      return;
    }

    this.kioskWindow = new BrowserWindow({
      fullscreen: true,
      kiosk: true,
      alwaysOnTop: true,
      frame: false,
      closable: false,
      skipTaskbar: true,
      webPreferences: {
        devTools: false,
        contextIsolation: true,
        sandbox: true,
        nodeIntegration: false,
      },
    });

    // Intercept and discard forbidden shortcuts
    this.kioskWindow.webContents.on('before-input-event', (_event, input) => {
      const shortcut = [
        input.alt ? 'Alt' : '',
        input.control ? 'Control' : '',
        input.shift ? 'Shift' : '',
        input.meta ? 'Meta' : '',
        input.key,
      ]
        .filter(Boolean)
        .join('+');

      if (BLOCKED_SHORTCUTS.includes(shortcut)) {
        _event.preventDefault();
      }
    });

    // Handle window close (should not happen, but be defensive)
    this.kioskWindow.on('closed', () => {
      this.kioskWindow = null;
    });

    // TODO: load the actual renderer; for now, use a data URL so we can
    // still show *something* even before the renderer is built.
    const initialHtml = this.renderKioskHtml(content);
    this.kioskWindow.loadURL(
      `data:text/html,${encodeURIComponent(initialHtml)}`,
    );

    this.showCount += 1;
  }

  hideKioskOverlay(): void {
    if (this.kioskWindow && !this.kioskWindow.isDestroyed()) {
      this.kioskWindow.hide();
      this.kioskWindow.destroy();
    }
    this.kioskWindow = null;
  }

  updateTimer(timeString: string): void {
    if (this.kioskWindow && !this.kioskWindow.isDestroyed()) {
      this.kioskWindow.webContents.send('overlay:timer', { timeString });
    }
  }

  sendAnnouncement(text: string, durationMs: number): void {
    if (this.kioskWindow && !this.kioskWindow.isDestroyed()) {
      this.kioskWindow.webContents.send('overlay:announcement', {
        text,
        durationMs,
      });
    }
  }

  async restartPC(): Promise<void> {
    await execAsync('shutdown /r /t 0');
  }

  async shutdownPC(): Promise<void> {
    await execAsync('shutdown /s /t 0');
  }

  async captureScreenshot(): Promise<Buffer> {
    const sources = await desktopCapturer.getSources({
      types: ['screen'],
      thumbnailSize: { width: 1280, height: 720 },
    });

    if (!sources || sources.length === 0) {
      throw new Error('No screen sources available for screenshot');
    }

    const primaryScreen = sources[0];
    let pngBuffer: Buffer;

    if (primaryScreen.thumbnail) {
      pngBuffer = primaryScreen.thumbnail.toPNG();
    } else {
      throw new Error('Screenshot thumbnail not available');
    }

    // Resize to max 1280x720 and compress as JPEG at 80% quality
    const compressed = await sharp(pngBuffer)
      .resize(1280, 720, { fit: 'inside', withoutEnlargement: true })
      .jpeg({ quality: 80 })
      .toBuffer();

    return compressed;
  }

  async enableAutoStart(): Promise<void> {
    // Use reg.exe to add an auto-start entry
    const command = `reg.exe add "${AUTO_START_KEY}" /v "${APP_NAME}" /d "${process.execPath}" /f`;
    await execAsync(command);
  }

  async disableAutoStart(): Promise<void> {
    const command = `reg.exe delete "${AUTO_START_KEY}" /v "${APP_NAME}" /f`;
    await execAsync(command);
  }

  async getSystemInfo(): Promise<SystemInfo> {
    const [cpu, mem, disk, osInfo] = await Promise.all([
      si.cpu(),
      si.mem(),
      si.diskLayout(),
      si.osInfo(),
    ]);

    const totalDisk = disk.reduce((acc, d) => acc + (d.size || 0), 0);

    return {
      cpuModel: cpu.brand,
      cpuCores: cpu.cores || os.cpus().length,
      totalMemoryGB: Math.floor(mem.total / 1024 / 1024 / 1024),
      totalDiskGB: Math.floor(totalDisk / 1024 / 1024 / 1024),
      osName: osInfo.platform,
      osVersion: osInfo.release,
      hostname: os.hostname(),
    };
  }

  /**
   * Render a basic HTML string for the kiosk overlay.
   *
   * This is a temporary fallback until the React renderer is built.
   * In production, the kiosk window will load the built renderer.html.
   */
  private renderKioskHtml(_content: OverlayContent): string {
    // Simple styled HTML with the cafe name and session indicator
```
